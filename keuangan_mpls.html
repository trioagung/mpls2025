const API_URL = "https://script.google.com/macros/s/AKfycbx.../exec"; // ganti dengan URL Web App kamu const transaksi = []; function tambahTransaksi() { const keterangan = document.getElementById("keterangan").value.trim(); const jumlah = parseFloat(document.getElementById("jumlah").value); const jenis = document.getElementById("jenis").value; if (!keterangan || isNaN(jumlah) || jumlah <= 0) { alert("Mohon isi data dengan benar."); return; } const formData = new URLSearchParams(); formData.append("keterangan", keterangan); formData.append("jumlah", jumlah); formData.append("jenis", jenis); fetch(API_URL, { method: "POST", body: formData }) .then(res => res.json()) .then(result => { alert("✅ " + result.pesan); transaksi.push({ tanggal: new Date().toLocaleDateString('id-ID'), keterangan, masuk: jenis === 'masuk' ? jumlah : 0, keluar: jenis === 'keluar' ? jumlah : 0 }); renderTabel(); document.getElementById("keterangan").value = ""; document.getElementById("jumlah").value = ""; }) .catch(err => alert("❌ Gagal: " + err)); }
